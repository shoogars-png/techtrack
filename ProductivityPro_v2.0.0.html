Créer le script Apps Script

Ouvrez votre fichier Sheets → menu Extensions → Apps Script, collez ce code :

function doGet(e) {
  const sheet = e.parameter.sheet || 'AGENTS';
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const ws = ss.getSheetByName(sheet);
    if (!ws) return ContentService.createTextOutput(JSON.stringify({error:'Onglet introuvable: '+sheet})).setMimeType(ContentService.MimeType.JSON);
    const data = ws.getDataRange().getValues();
    return ContentService.createTextOutput(JSON.stringify({rows:data})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({error:err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
